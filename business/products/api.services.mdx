## Access Protocols

### 1. REST API

```bash
# Standard HTTP requests
curl https://api.services/v1/functions/send-email \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -d '{
    "to": "user@example.com",
    "subject": "Hello",
    "body": "Message content"
  }'
```

### 2. Model Context Protocol (MCP)

```typescript
// AI agents use MCP tools
const result = await mcp.call('send-email', {
  to: 'user@example.com',
  subject: 'Hello',
  body: 'Message content'
})
```

### 3. Workers RPC (Service Bindings)

```typescript
// Direct RPC calls from Workers
const result = await env.API_SERVICES.sendEmail({
  to: 'user@example.com',
  subject: 'Hello',
  body: 'Message content'
})
```

### 4. Type-Safe SDKs

```typescript
import { functions } from '@dot-do/api-services'

const result = await functions.sendEmail({
  to: 'user@example.com',
  subject: 'Hello',
  body: 'Message content'
})
```

### Service Bindings

```typescript
// Internal RPC communication
export class FunctionService extends WorkerEntrypoint {
  async execute(name: string, input: any) {
    // Validate input
    const validated = await this.env.DB.validate(name, input)

    // Execute function
    const result = await this[name](validated)

    // Log to analytics
    await this.env.ANALYTICS.log('function_execution', {
      name, duration, success: true
    })

    return result
  }
}
```
